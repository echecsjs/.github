Clean APIs, strict types

Every package is TypeScript-first with full type inference

import { expected, update } from '@echecs/elo';

// Two players both rated 1400. Player A wins.
const [playerA, playerB] = update(1400, 1400, 1);

console.log(playerA); // 1410
console.log(playerB); // 1390

// Win probability
const probability = expected(1400, 1800);

console.log(probability); // 0.09
